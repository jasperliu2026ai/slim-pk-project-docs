## 一、方案选择：MobX-miniprogram

### 为什么用 MobX

| 对比项 | 手写全局 store（globalData） | MobX-miniprogram |
|--------|------------------------------|------------------|
| 响应式更新 | 手动 setData，容易遗漏 | 自动 reaction，绑定页面 |
| TypeScript 支持 | 无类型 | 完整类型推导 |
| 跨页通信 | 需要 EventBus 或全局事件 | Observable，自动同步 |
| 代码量 | 少（无依赖） | 略多，但可维护性高 |

### 安装

```bash
npm install mobx-miniprogram mobx-miniprogram-bindings
```

## 三、UserStore

```typescript
// store/user.ts
import { makeAutoObservable } from 'mobx-miniprogram';

class UserStore {
  userId = '';
  nickname = '';
  avatarUrl = '';
  token = '';
  isLoggedIn = false;
  privacyConsented = false; // 敏感信息同意状态

  constructor() {
    makeAutoObservable(this);
  }

  // 从 wx.Storage 恢复登录态
  restoreFromStorage() {
    const token = wx.getStorageSync('token') as string;
    const userInfo = wx.getStorageSync('userInfo') as typeof this | null;
    if (token && userInfo) {
      this.token = token;
      this.userId = userInfo.userId;
      this.nickname = userInfo.nickname;
      this.avatarUrl = userInfo.avatarUrl;
      this.privacyConsented = wx.getStorageSync('privacyConsented') as boolean;
      this.isLoggedIn = true;
    }
  }

  // 登录后保存
  setUserInfo(data: {
    token: string;
    userId: string;
    nickName: string;
    avatarUrl: string;
  }) {
    this.token = data.token;
    this.userId = data.userId;
    this.nickname = data.nickName;
    this.avatarUrl = data.avatarUrl;
    this.isLoggedIn = true;
    wx.setStorageSync('token', data.token);
    wx.setStorageSync('userInfo', {
      userId: data.userId,
      nickname: data.nickName,
      avatarUrl: data.avatarUrl,
    });
  }

  // 隐私同意
  setPrivacyConsented(value: boolean) {
    this.privacyConsented = value;
    wx.setStorageSync('privacyConsented', value);
  }

  // 退出 / 删除账号
  logout() {
    this.token = '';
    this.userId = '';
    this.nickname = '';
    this.avatarUrl = '';
    this.isLoggedIn = false;
    this.privacyConsented = false;
    wx.clearStorageSync();
  }
}

export const userStore = new UserStore();
```

## 四、PkStore

```typescript
// store/pk.ts
import { makeAutoObservable } from 'mobx-miniprogram';
import type { PkListItem, RankingData } from '../typings/pk';

class PkStore {
  // 首页 PK 列表
  ongoingList: PkListItem[] = [];
  endedList: PkListItem[] = [];
  listLoading = false;

  // 当前查看的 PK
  currentPkId = '';
  currentRanking: RankingData | null = null;
  rankingLoading = false;
  rankingLastUpdatedAt = '';

  constructor() {
    makeAutoObservable(this);
  }

  setList(ongoing: PkListItem[], ended: PkListItem[]) {
    this.ongoingList = ongoing;
    this.endedList = ended;
  }

  setCurrentPk(pkId: string) {
    this.currentPkId = pkId;
    this.currentRanking = null; // 切换 PK 清空排行榜缓存
  }

  setRanking(data: RankingData) {
    this.currentRanking = data;
    this.rankingLastUpdatedAt = data.lastUpdatedAt;
  }

  // 首页打卡状态更新（打卡成功后，不用重新拉列表）
  markTodayCheckinDone(pkId: string) {
    const pk = this.ongoingList.find(p => p.pkId === pkId);
    if (pk) pk.myCheckedInToday = true;
  }
}

export const pkStore = new PkStore();
```

## 五、CheckinStore

```typescript
// store/checkin.ts
import { makeAutoObservable } from 'mobx-miniprogram';

interface TodayCheckin {
  done: boolean;
  weight: number | null;
  photoUrl: string | null;
  dietDone: boolean | null;
  exerciseDone: boolean | null;
}

class CheckinStore {
  // 今日打卡状态（按 pkId 存储）
  private todayMap: Record<string, TodayCheckin> = {};

  constructor() {
    makeAutoObservable(this);
  }

  getToday(pkId: string): TodayCheckin {
    return this.todayMap[pkId] ?? {
      done: false,
      weight: null,
      photoUrl: null,
      dietDone: null,
      exerciseDone: null,
    };
  }

  setTodayWeight(pkId: string, weight: number, photoUrl: string) {
    const today = this.getToday(pkId);
    this.todayMap[pkId] = { ...today, weight, photoUrl, done: true };
  }

  setTodayLifestyle(pkId: string, dietDone: boolean, exerciseDone: boolean) {
    const today = this.getToday(pkId);
    this.todayMap[pkId] = { ...today, dietDone, exerciseDone };
  }

  // 每天 00:00 清空（由 App 生命周期检测日期变化触发）
  clearTodayForNewDay() {
    this.todayMap = {};
  }
}

export const checkinStore = new CheckinStore();
```

## 六、页面绑定（示例）

```typescript
import { ComponentWithStore } from 'mobx-miniprogram-bindings';
import { pkStore, userStore } from '../../store/index';

ComponentWithStore({
  storeBindings: [
    {
      store: pkStore,

      actions: ['setRanking'],

    {
      store: userStore,
      fields: ['userId'],

  async onLoad(options) {
    const { pkId } = options;
```

## 七、数据持久化策略

| 数据 | 存储方式 | 说明 |
|------|----------|------|
| JWT token | `wx.setStorageSync('token')` | 登录时写，退出时清 |
| 用户基本信息 | `wx.setStorageSync('userInfo')` | 非敏感，可本地缓存 |
| 隐私同意状态 | `wx.setStorageSync('privacyConsented')` | 避免每次重启重弹弹框 |
| PK 列表 / 排行榜 | 仅内存（MobX store） | 每次进页面重新拉接口 |
| 体重数据 | **不本地持久化** | 健康敏感数据，不落 localStorage |
| 打卡照片 | **临时文件，不持久化** | 上传后即丢弃 tempFilePath |

> ⚠️ 体重等健康数据**绝对不写入 wx.setStorageSync**，只在页面级变量和 MobX store（内存）中流转，提交后立即丢弃本地副本。
