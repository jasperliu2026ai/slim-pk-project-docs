## 一、微信登录流程

### 完整时序

```
用户                  小程序前端                  后端
 │                        │                        │
 │── 打开小程序 ──────────→│                        │
 │                        │── wx.login() ──────────→ code
 │                        │← code ─────────────────│
 │                        │                        │
 │                        │── POST /auth/login ────→│
 │                        │   { code }              │── code2session ──→ 微信服务器
 │                        │                        │←── openid / unionid ──────────
 │                        │                        │── 生成 accessToken ──────────
 │                        │← { accessToken, ... } ─│
 │                        │                        │
 │                  存入 Storage + userStore         │
 │                        │                        │
 │ （首次用户）             │                        │
 │                        │── 展示隐私政策弹框 ─────→│
 │── 同意 ────────────────→│                        │
 │                        │── POST /user/privacy ──→│
```

### 代码实现

```typescript
// services/auth.ts
export async function login(): Promise<void> {
  // 1. 获取 wx.login code
  const { code } = await new Promise<WechatMiniprogram.LoginSuccessCallbackResult>(
    (resolve, reject) =>
      wx.login({ success: resolve, fail: reject })
  );

  // 2. 换取 accessToken
  const res = await request<LoginData>({
    url: '/auth/login',
    method: 'POST',
    data: { code },
  });

  // 3. 存储登录态
  userStore.login(res.data);

  // 4. 首次用户：展示隐私弹框
  if (res.data.isNewUser && !userStore.privacyAccepted) {
    await showPrivacyModal();
  }
}

// 隐私弹框（必须同意才能继续）
function showPrivacyModal(): Promise<void> {
  return new Promise((resolve) => {
    wx.showModal({
      title: '隐私保护提示',
      content:
        '本小程序会收集您的体重、运动等健康数据，用于 PK 积分计算，不会对外分享。请阅读《隐私政策》后继续使用。',
      confirmText: '同意并继续',
      cancelText: '不同意',
      success(res) {
        if (res.confirm) {
          userStore.acceptPrivacy();
          resolve();
        } else {
          // 不同意则退出小程序
          wx.exitMiniProgram();
        }
      },
    });
  });
}
```

## 二、打卡图水印合成

### 实现方案

```typescript
// utils/watermark.ts

interface WatermarkOptions {
  imageUrl: string;         // 原始拍照临时路径
  nickname: string;
  checkinDate: string;      // YYYY-MM-DD
  checkinTime: string;      // HH:mm
}

export async function addWatermark(options: WatermarkOptions): Promise<string> {
  const { imageUrl, nickname, checkinDate, checkinTime } = options;

  // 1. 创建 Canvas 2D（推荐用 type='2d'，兼容性更好）
  const canvasId = 'watermark-canvas';
  const canvas = await getCanvas(canvasId);
  const ctx = canvas.getContext('2d');

  // 2. 获取图片信息（宽高）
  const imgInfo = await new Promise<WechatMiniprogram.GetImageInfoSuccessCallbackResult>(
    (resolve, reject) =>
      wx.getImageInfo({ src: imageUrl, success: resolve, fail: reject })
  );

  // 3. 设置 Canvas 尺寸（保持图片比例）
  const W = 750;  // rpx → px 已在调用前转换
  const H = Math.round(W * imgInfo.height / imgInfo.width);
  canvas.width = W;
  canvas.height = H;

  // 4. 绘制原图
  const img = canvas.createImage();
  img.src = imageUrl;
  await new Promise<void>((resolve) => { img.onload = () => resolve(); });
  ctx.drawImage(img, 0, 0, W, H);

  // 5. 绘制水印条（底部黑色半透明）
  const barHeight = 80;
  ctx.fillStyle = 'rgba(0, 0, 0, 0.55)';
  ctx.fillRect(0, H - barHeight, W, barHeight);

  // 6. 绘制文字
  ctx.fillStyle = '#FFFFFF';
  ctx.font = 'bold 28px sans-serif';
  ctx.fillText(`${checkinDate} ${checkinTime}`, 20, H - barHeight + 30);

  ctx.font = '24px sans-serif';
  ctx.fillText(`@${nickname} · 减肥PK小程序`, 20, H - barHeight + 60);

  // 7. 导出临时文件
  const result = await new Promise<string>((resolve, reject) =>
    wx.canvasToTempFilePath({
      canvas,
      fileType: 'jpg',
      quality: 0.9,
      success: (res) => resolve(res.tempFilePath),
      fail: reject,
    })
  );

  return result;
}

function getCanvas(id: string): Promise<WechatMiniprogram.Canvas> {
  return new Promise((resolve) => {
    wx.createSelectorQuery()
      .select(`#${id}`)
      .fields({ node: true, size: true })
      .exec((res) => resolve(res[0].node));
  });
}
```

// components/share-poster/share-poster.ts
import type { SettlementData } from '../../types/pk';

Component({
  methods: {
    async generateAndSave(settlement: SettlementData) {
      wx.showLoading({ title: '生成海报中...' });
      try {
        const filePath = await this.drawPoster(settlement);
        await this.saveToAlbum(filePath);
        wx.showToast({ title: '已保存到相册', icon: 'success' });
      } catch (e) {
        wx.showToast({ title: '生成失败，请重试', icon: 'none' });
      } finally {
        wx.hideLoading();
      }
    },

    async drawPoster(data: SettlementData): Promise<string> {
      const canvas = await getCanvas('poster-canvas');
      const ctx = canvas.getContext('2d');
      const W = 750, H = 1334;
      canvas.width = W;
      canvas.height = H;

      // 1. 渐变背景
      const gradient = ctx.createLinearGradient(0, 0, 0, H);
      gradient.addColorStop(0, '#FF6B35');
      gradient.addColorStop(1, '#C44A1A');
      ctx.fillStyle = gradient;
      ctx.fillRect(0, 0, W, H);

      // 2. 头像（圆形裁剪）
      await drawCircleAvatar(ctx, data.avatarUrl, 315, 320, 60);

      // 3. 文字
      ctx.fillStyle = '#FFFFFF';
      ctx.font = 'bold 40px sans-serif';
      ctx.textAlign = 'center';
      ctx.fillText(`「${data.pkName}」`, W / 2, 200);

      ctx.font = '32px sans-serif';
      ctx.fillText(data.nickname, W / 2, 460);

      ctx.font = 'bold 56px sans-serif';
      ctx.fillText(`🏆 第 ${data.myRank}/${data.totalMembers} 名`, W / 2, 540);

      ctx.font = '36px sans-serif';
      ctx.fillText(`减重 ${data.weightLoss.toFixed(1)} kg`, W / 2, 620);
      ctx.fillText(`积分 ${data.myScore} 分`, W / 2, 670);

      // 4. 分割线
      ctx.strokeStyle = 'rgba(255,255,255,0.4)';
      ctx.lineWidth = 1;
      ctx.beginPath();
      ctx.moveTo(60, 900);
      ctx.lineTo(W - 60, 900);
      ctx.stroke();

      // 5. 小程序码
      if (data.qrCodeUrl) {
        await drawImage(ctx, data.qrCodeUrl, 295, 940, 160, 160);
      }

      ctx.font = '28px sans-serif';
      ctx.fillStyle = 'rgba(255,255,255,0.8)';
      ctx.fillText('扫码加入下一场 PK', W / 2, 1130);

      // 6. 导出
      return new Promise((resolve, reject) =>
        wx.canvasToTempFilePath({
          canvas,
          fileType: 'jpg',
          quality: 0.95,
          success: (res) => resolve(res.tempFilePath),
          fail: reject,
        })
      );
    },

    async saveToAlbum(filePath: string) {
      // 检查相册权限
      const auth = await new Promise<WechatMiniprogram.AuthSetting>(
        (resolve) => wx.getSetting({ success: (res) => resolve(res.authSetting) })
      );

      if (auth['scope.writePhotosAlbum'] === false) {
        // 用户之前拒绝了，引导去设置
        const { confirm } = await showConfirm('需要相册权限才能保存，去设置页开启？');
        if (confirm) wx.openSetting();
        return;
      }

      await new Promise<void>((resolve, reject) =>
        wx.saveImageToPhotosAlbum({
          filePath,
          success: () => resolve(),
          fail: reject,
        })
      );
    },
  },
});

## 四、积分公式（前端展示版）

```typescript
// utils/score.ts
// ⚠️ 此处公式仅用于前端本地预估/动效展示
// 权威积分由后端计算，前端不作为结算依据

interface DailyCheckinParams {
  checkedIn: boolean;
  weightRecorded: boolean;
  photoUploaded: boolean;
  exerciseLogged: boolean;
  consecutiveDays: number;
}

export function calcDailyScore(params: DailyCheckinParams): number {
  if (!params.checkedIn) return 0;

  let base = 10; // 打卡基础分

  if (params.weightRecorded) base += 5;
  if (params.photoUploaded) base += 5;
  if (params.exerciseLogged) base += 5;

  // 连续打卡奖励
  const bonus = calcConsecutiveBonus(params.consecutiveDays);

  return base + bonus;
}

function calcConsecutiveBonus(days: number): number {
  if (days >= 21) return 15;
  if (days >= 14) return 10;
  if (days >= 7)  return 5;
  if (days >= 3)  return 2;
  return 0;
}
```

## 六、极端数据安全拦截

```typescript
// utils/validate.ts
export interface WeightValidation {
  valid: boolean;
  error?: string;
  warning?: string;
}

export function validateWeight(weight: number): WeightValidation {
  if (isNaN(weight) || weight <= 0) {
    return { valid: false, error: '请输入有效的体重' };
  }
  if (weight < 20 || weight > 500) {
    return { valid: false, error: '体重超出合理范围（20-500 kg）' };
  }
  return { valid: true };
}

export function validateBMI(height: number, weight: number): WeightValidation {
  const bmi = weight / Math.pow(height / 100, 2);
  if (bmi < 18.5) {
    return {
      valid: true,
      warning: 'BMI 偏低，建议先咨询医生再制定减重目标',
    };
  }
  return { valid: true };
}
```
