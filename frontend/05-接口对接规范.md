## 一、request.ts 封装

```typescript
// services/request.ts

const BASE_URL = getApp().globalData.apiBaseUrl; // 通过 app.ts 注入，区分 dev/prod

interface RequestOptions {
  url: string;
  method?: 'GET' | 'POST' | 'PUT' | 'DELETE';
  data?: Record<string, any>;
  showLoading?: boolean;
  loadingText?: string;
}

interface ApiResponse<T = any> {
  code: number;
  message: string;
  data: T;
}

export function request<T = any>(options: RequestOptions): Promise<ApiResponse<T>> {
  const {
    url,
    method = 'GET',
    data,
    showLoading = false,
    loadingText = '加载中...',
  } = options;

  if (showLoading) {
    wx.showLoading({ title: loadingText, mask: true });
  }

  const token = getApp().globalData.accessToken;

  return new Promise((resolve, reject) => {
    wx.request({
      url: `${BASE_URL}${url}`,
      method,
      data,
      header: {
        'Content-Type': 'application/json',
        ...(token ? { Authorization: `Bearer ${token}` } : {}),
      },
      success(res) {
        if (showLoading) wx.hideLoading();

        const body = res.data as ApiResponse<T>;

        if (res.statusCode === 401) {
          // token 过期，跳转登录
          handleUnauthorized();
          reject(new Error('UNAUTHORIZED'));
          return;
        }

        if (res.statusCode >= 500) {
          wx.showToast({ title: '服务繁忙，请稍后重试', icon: 'none' });
          reject(new Error('SERVER_ERROR'));
          return;
        }

        if (body.code !== 0) {
          // 业务错误，由调用方决定是否提示
          reject(body);
          return;
        }

        resolve(body);
      },
      fail(err) {
        if (showLoading) wx.hideLoading();
        wx.showToast({ title: '网络异常，请检查网络', icon: 'none' });
        reject(err);
      },
    });
  });
}

function handleUnauthorized() {
  const { userStore } = require('../store/index');
  userStore.logout();
  wx.reLaunch({ url: '/pages/login/index' });
}
```

// app.ts
App({

## 三、接口清单

### 3.1 认证模块（auth.ts）

#### POST `/auth/login`
微信登录换取 accessToken

**请求体**
```typescript
{
  code: string;          // wx.login() 返回的 code
  nickName?: string;     // 用户授权的昵称（首次登录可能有）
  avatarUrl?: string;    // 用户授权的头像
}
```

**响应体**
```typescript
{
  code: 0,
  data: {
    userId: string;
    openId: string;
    nickname: string;
    avatarUrl: string;
    accessToken: string;
    expireAt: number;      // Unix 时间戳（毫秒）
    isNewUser: boolean;
  }
}
```

**错误码**

| code | 含义 |
|------|------|
| 1001 | code 无效或已过期 |
| 1002 | 服务端 code2session 失败 |

---

**响应体**：同 login 的 data 结构

### 3.3 PK 房间模块（pk.ts）

#### POST `/pk/create`
创建 PK 房间

**请求体**
```typescript
{
  name: string;               // PK 名称，1-20 字
  targetWeight: number;       // 目标体重（kg），精度 0.1
  currentWeight: number;      // 当前体重（kg），精度 0.1
  durationDays: 7 | 14 | 21 | 30;
  maxMembers: number;         // 最多人数 2-10
}
```

**响应体**
```typescript
{
  code: 0,
  data: {
    pkId: string;
    inviteCode: string;       // 6位邀请码
    inviteUrl: string;        // 小程序分享路径（不含域名）
    qrCodeUrl: string;        // 邀请二维码图片 URL
  }
}
```

**错误码**

| code | 含义 |
|------|------|
| 2001 | 名称含违禁词 |
| 2002 | 体重参数非法 |
| 2003 | 用户已有进行中的 PK 超过上限（暂定3场）|

### 3.4 打卡模块（checkin.ts）

#### POST `/checkin`
提交打卡

**请求体（multipart/form-data）**
```typescript
{
  pkId: string;
  weight: number;               // 当日体重（kg），精度 0.1
  imageUrl: string;             // 水印合成后已上传 OSS 的图片 URL
  exerciseType?: string;        // 运动类型（跑步/游泳/骑行/其他）
  exerciseDurationMin?: number; // 运动时长（分钟）
  checkinAt: string;            // ISO8601 时间戳（前端本地时间）
}
```

> **注意**：图片先上传到 OSS，获取 URL 后再调此接口。OSS 直传方案由运维/后端提供上传凭证接口。

**响应体**
```typescript
{
  code: 0,
  data: {
    checkinId: string;
    newScore: number;            // 打卡后最新积分
    scoreDelta: number;          // 本次得分增量
    consecutiveDays: number;     // 当前连续打卡天数
    bonusApplied: boolean;       // 是否触发了连续奖励
  }
}
```

**错误码**

| code | 含义 |
|------|------|
| 3001 | 今日已打卡 |
| 3002 | PK 未开始或已结束 |
| 3003 | 体重超出合理范围（<20kg 或 >500kg）|
| 3004 | 图片 URL 无效 |

## 四、错误处理规范

### 4.1 错误分层

| 层 | 错误类型 | 处理方式 |
|----|----------|----------|
| 网络层 | 无网络、超时 | `request.ts` 统一 toast 提示 |
| HTTP 层 | 4xx / 5xx | `request.ts` 处理 401/5xx，其余抛给调用方 |
| 业务层 | code !== 0 | 各页面/服务根据错误码处理 |
| 已知业务错误 | 如 3001「今日已打卡」| 前端展示友好文案，不视为「错误」 |

### 4.2 错误码对照表（前端关心的）

| code | 含义 | 前端处理 |
|------|------|----------|
| 0 | 成功 | 正常处理 |
| 401 | 未授权 | 清空登录态，跳登录页 |
| 1001 | code 无效 | toast 提示，重新触发 wx.login |
| 2001 | 名称含违禁词 | 表单内联错误提示 |
| 2011 | PK 已满员 | toast 提示 |
| 2013 | 已是成员 | 直接跳转 pk-detail |
| 3001 | 今日已打卡 | 展示「今日已打卡」状态，不报错 |
| 3003 | 体重超出范围 | 表单内联错误提示（理论上前端已拦截）|

### 4.3 防重复提交

```typescript
// 打卡提交防重复
let submitting = false;

async function submitCheckin() {
  if (submitting) return;
  submitting = true;
  this.setData({ submitLoading: true });
  try {
    await checkinApi.submit(payload);
    // 成功逻辑
  } finally {
    submitting = false;
    this.setData({ submitLoading: false });
  }
}
```
